## MOV OPERATIONS
- 0x10 - MOV REG, VALUE
- 0x11 - MOV REG, REG

## MATH OPERATIONS
- 0x200 - ADD REG, REG
- 0x201 - ADD REG, VALUE
- 0x202 - ADD VALUE, VALUE

- 0x210 - SUB REG, REG
- 0x211 - SUB REG, VALUE
- 0x212 - SUB VALUE, VALUE

- 0x220 - MUL REG, REG
- 0x221 - MUL REG, VALUE
- 0x222 - MUL VALUE, VALUE

- 0x230 - DIV REG, REG
- 0x231 - DIV REG, VALUE
- 0x331 - DIV VALUE, VALUE

## LOGICAL OPERATIONS
- 0x300 - CMP REG, REG
- 0x301 - CMP REG, VALUE
- 0x302 - CMP VALUE, VALUE

- 0x310 - ZERO VALUE
- 0x311 - ZERO REG

- 0x320 - GT REG, REG
- 0x321 - GT REG, VALUE
- 0x322 - GT VALUE, VALUE

- 0x320 - LT REG, REG
- 0x321 - LT REG, VALUE
- 0x322 - LT VALUE, VALUE

## FLOW CONTROL
- 0x40 - LABEL ID
- 0x41 - JMP ID
- 0x42 - JE ID
- 0x43 - GT ID
- 0x44 - LT ID
- 0x45 - JZ ID
- 0x46 - JNE ID

## MEMORY MANAGEMENT
- 0x50 - STORE ADDR, REG
- 0x51 - STORE ADDR, VALUE
- 0x52 - LOAD ADDR
